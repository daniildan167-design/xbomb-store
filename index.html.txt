    // 5. ГЕНЕРАЦИЯ ЦЕН И СКИДОК (в BYN)

    function generatePrice() {
        const base = 19.90 + Math.random() * 130;
        return Math.round(base / 0.10) * 0.10;
    }

    function getPriceWithDiscount(price) {
        const discount = Math.random() * 0.45;
        const discounted = price * (1 - discount);
        return {
            current: parseFloat(discounted.toFixed(2)),
            old: price,
            discountPercent: Math.round(discount * 100)
        };
    }

    // ------------------------------------------------------------
    // 6. ЗАГРУЗКА ВСЕХ ИГР (пагинация по страницам)
    // ------------------------------------------------------------
    async function loadGames(searchQuery = '') {
        gamesGrid.innerHTML = `<div class="loading-spinner"><div class="spinner"></div><p>Загрузка каталога игр...</p></div>`;
        let allLoaded = [];
        let page = 1;
        const perPage = 40;
        let baseUrl =
            `https://api.rawg.io/api/games?key=${CONFIG.RAWG_API_KEY}&platforms=1,186&page_size=${perPage}`;

        if (searchQuery.trim()) {
            baseUrl += `&search=${encodeURIComponent(searchQuery)}`;
            // при поиске только первая страница
            try {
                const resp = await fetch(baseUrl);
                if (!resp.ok) throw new Error(`API error ${resp.status}`);
                const data = await resp.json();
                allLoaded = data.results || [];
            } catch (e) {
                console.error(e);
                showError();
                return;
            }
        } else {
            // загружаем все страницы, максимум 2000 игр
            let hasMore = true;
            while (hasMore && allLoaded.length < 2000) {
                const url = `${baseUrl}&page=${page}`;
                try {
                    const resp = await fetch(url);
                    if (!resp.ok) throw new Error(`API error ${resp.status}`);
                    const data = await resp.json();
                    const results = data.results || [];
                    if (results.length === 0) break;
                    allLoaded = [...allLoaded, ...results];
                    page++;
                    if (allLoaded.length >= (data.count || 0) || allLoaded.length >= 2000) {
                        hasMore = false;
                    }
                } catch (e) {
                    console.error(e);
                    hasMore = false;
                }
            }
        }

        if (allLoaded.length === 0) {
            showError('Игры не найдены');
            return;
        }

        // Обогащаем данными
        allGames = allLoaded.map(game => {
            const price = generatePrice();
            const priceData = getPriceWithDiscount(price);
            const genres = (game.genres || []).map(g => ({
                name: g.name,
                slug: g.slug,
                name_ru: genreMap[g.slug] || g.name
            }));
            return {
                ...game,
                price: priceData.current,
                oldPrice: priceData.old,
                discount: priceData.discountPercent,
                genres_ru: genres,
                description_ru: getRussianDescription(game, game.description_raw),
                isNew: isNewGame(game.released),
                isPreorder: isPreorderGame(game.released),
            };
        });

        // Сортируем: предзаказы и новинки в начало
        allGames.sort((a, b) => {
            if (a.isPreorder && !b.isPreorder) return -1;
            if (!a.isPreorder && b.isPreorder) return 1;
            if (a.isNew && !b.isNew) return -1;
            if (!a.isNew && b.isNew) return 1;
            return 0;
        });

        applyFilters();
    }

                </button>
            </div>
        `;
        pagination.innerHTML = '';
    }

    // 7. ОПРЕДЕЛЕНИЕ НОВИНОК И ПРЕДЗАКАЗОВ

    function isNewGame(released) {
        if (!released) return false;
        const date = new Date(released);
        const now = new Date();
        const diff = (now - date) / (1000 * 60 * 60 * 24);
        return diff >= 0 && diff < 30;
    }

    function isPreorderGame(released) {
        if (!released) return false;
        const date = new Date(released);
        const now = new Date();
        const diff = (date - now) / (1000 * 60 * 60 * 24);
        return diff > 0 && diff < 60;
    }

    // ------------------------------------------------------------
    // 9. ФИЛЬТРАЦИЯ И СОРТИРОВКА
    // ------------------------------------------------------------
    function applyFilters() {
        const search = searchInput.value.trim().toLowerCase();
        const genre = genreFilter.value;
        const sort = sortSelect.value;

        let filtered = [...allGames];

        if (search) {
            filtered = filtered.filter(g => g.name.toLowerCase().includes(search));
        }
        if (genre !== 'all') {
            filtered = filtered.filter(g =>
                g.genres_ru.some(gr => gr.slug === genre || gr.name_ru.toLowerCase() === genre)
            );
        }

        switch (sort) {
            case 'popular':
                filtered.sort((a, b) => (b.metacritic || 0) - (a.metacritic || 0));
                break;
            case 'new':
                filtered.sort((a, b) => {
                    const da = a.released ? new Date(a.released) : new Date(0);
                    const db = b.released ? new Date(b.released) : new Date(0);
                    return db - da;
                });
                break;
            case 'price-asc':
                filtered.sort((a, b) => a.price - b.price);
                break;
            case 'price-desc':
                filtered.sort((a, b) => b.price - a.price);
                break;
            default: break;
        }

        filteredGames = filtered;
        resultsCount.innerHTML = `Найдено: <strong>${filtered.length}</strong> игр`;
        currentPage = 1;
        renderPage();
    }

    // ------------------------------------------------------------
    // 10. ОТРИСОВКА СТРАНИЦЫ
    // ------------------------------------------------------------
    function renderPage() {
        const total = filteredGames.length;
        const totalPages = Math.ceil(total / pageSize) || 1;
        const start = (currentPage - 1) * pageSize;
        const end = Math.min(start + pageSize, total);
        const pageGames = filteredGames.slice(start, end);

        if (pageGames.length === 0) {
            gamesGrid.innerHTML =
                `<div style="grid-column:1/-1; text-align:center; padding:40px; color:#6a8a8a;">🔍 Ничего не найдено</div>`;
        } else {
            gamesGrid.innerHTML = pageGames.map(game => renderCard(game)).join('');
        }

        // Пагинация
        let html = '';
        if (totalPages > 1) {
            html += `<button onclick="goToPage(1)" ${currentPage===1?'disabled':''}>«</button>`;
            for (let i = 1; i <= Math.min(totalPages, 7); i++) {
                if (i === currentPage) {
                    html += `<button class="active">${i}</button>`;
                } else if (i === 1 || i === totalPages || Math.abs(i - currentPage) <= 2) {
                    html += `<button onclick="goToPage(${i})">${i}</button>`;
                } else if (i === 2 || i === totalPages - 1) {
                    html += `<span style="color:#4a5a5a;padding:0 4px;">…</span>`;
                }
            }
            html += `<button onclick="goToPage(${totalPages})" ${currentPage===totalPages?'disabled':''}>»</button>`;
        }
        pagination.innerHTML = html;
    }

    function goToPage(page) {
        currentPage = page;
        renderPage();
        document.querySelector('.container').scrollIntoView({ behavior: 'smooth', block: 'start' });
    }

    function renderCard(game) {
        const cover = game.background_image || 'https://via.placeholder.com/300x400/0b0e11/2a3a3e?text=No+Image';
        const year = game.released ? new Date(game.released).getFullYear() : '—';
        const rating = game.metacritic ? `⭐ ${game.metacritic}` : '';
        const genreNames = game.genres_ru.slice(0, 2).map(g => g.name_ru).join(', ') || '—';

        let badges = `<span class="badge badge-xbox">Xbox</span>`;
        if (game.isPreorder) badges += `<span class="badge badge-preorder">Предзаказ</span>`;
        if (game.isNew) badges += `<span class="badge badge-new">Новинка</span>`;
        if (game.discount > 15) badges += `<span class="badge badge-discount">-${game.discount}%</span>`;

        // Цена в выбранной валюте
        const priceConverted = convertCurrency(game.price);
        const oldConverted = convertCurrency(game.oldPrice);
        const symbol = CURRENCY_SYMBOLS[currentCurrency];

        return `
            <div class="game-card" onclick="openModal(${game.id})">
                <div class="img-wrap">
                    <img src="${cover}" alt="${game.name}" loading="lazy" onerror="this.src='https://via.placeholder.com/300x400/0b0e11/2a3a3e?text=Error'">
                    <div class="badges">${badges}</div>
                </div>
                <div class="info">
                    <h3>${game.name}</h3>
                    <div class="meta">
                        <span>${year}</span>
                        <span>${genreNames}</span>
                    </div>
                    ${rating ? `<div class="rating">${rating}</div>` : ''}
                    <div class="price-wrap">
                        <div class="price">
                            ${game.discount > 0 ? `<span class="old">${oldConverted.toFixed(2)} ${symbol}</span>` : ''}
                            ${priceConverted.toFixed(2)} ${symbol}
                        </div>
                    </div>
                </div>
            </div>
        `;
    }

    // 14. ОТПРАВКА ЗАКАЗА В TELEGRAM
    // ------------------------------------------------------------
    async function sendOrder(event, gameId) {
        event.preventDefault();
        const nameInput = document.getElementById('userName');
        const phoneInput = document.getElementById('userPhone');
        const statusDiv = document.getElementById('formStatus');
        const buyBtn = document.getElementById('buyBtn');

        const name = nameInput.value.trim();
        const phone = phoneInput.value.trim();
        if (!name || !phone) {
            statusDiv.className = 'form-status error';
            statusDiv.textContent = '⚠️ Заполните все поля';
            return;
        }

        const game = allGames.find(g => g.id === gameId);
        const gameName = game ? game.name : 'Игра';
        const price = game ? game.price : 0;
        const priceConv = convertCurrency(price);
        const symbol = CURRENCY_SYMBOLS[currentCurrency];
        const priceStr = `${priceConv.toFixed(2)} ${symbol}`;

        buyBtn.disabled = true;
        buyBtn.textContent = '⏳ Отправка...';
        statusDiv.className = '';

        try {
            const message =
                `🆕 НОВЫЙ ЗАКАЗ XBomb!\n\n` +
                `🎮 Игра: ${gameName}\n` +
                `💰 Цена: ${priceStr}\n` +
                `👤 Имя: ${name}\n` +
                `📞 Телефон: ${phone}\n` +
                `🕐 Время: ${new Date().toLocaleString('ru-RU')}`;

            const resp = await fetch(`https://api.telegram.org/bot${CONFIG.BOT_TOKEN}/sendMessage`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    chat_id: CONFIG.CHAT_ID,
                    text: message,
                }),
            });
            const result = await resp.json();
            if (result.ok) {
                statusDiv.className = 'form-status success';
                statusDiv.textContent = '✅ Заявка отправлена! Мы свяжемся с вами.';
                nameInput.value = '';
                phoneInput.value = '';
            } else {
                throw new Error(result.description || 'Ошибка');
            }
        } catch (e) {
            statusDiv.className = 'form-status error';
            statusDiv.textContent = '❌ Ошибка отправки. Попробуйте позже.';
        } finally {
            buyBtn.disabled = false;
            buyBtn.textContent = '📩 Отправить заявку';
        }
    }

    // ------------------------------------------------------------
    // 15. СОБЫТИЯ
    // ------------------------------------------------------------
    searchBtn.addEventListener('click', applyFilters);
    searchInput.addEventListener('keydown', (e) => { if (e.key === 'Enter') applyFilters(); });
    sortSelect.addEventListener('change', applyFilters);
    genreFilter.addEventListener('change', applyFilters);

    document.querySelectorAll('#genreNav a').forEach(el => {
        el.addEventListener('click', function() {
            document.querySelectorAll('#genreNav a').forEach(a => a.classList.remove('active'));
            this.classList.add('active');
            const genre = this.dataset.genre;
            genreFilter.value = genre === 'all' ? 'all' : genre;
            applyFilters();
        });
    });

    function resetFilters() {
        searchInput.value = '';
        sortSelect.value = 'popular';
        genreFilter.value = 'all';
        document.querySelectorAll('#genreNav a').forEach(a => a.classList.remove('active'));
        document.querySelector('#genreNav a[data-genre="all"]').classList.add('active');
        applyFilters();
    }

    // ------------------------------------------------------------
    // 16. ЗАПУСК
    // ------------------------------------------------------------
    loadGames();

    console.log('🎮 XBomb — магазин игр Xbox и PC');
    console.log('✅ Валюта, скидки, рейтинги, русские описания — всё работает');
