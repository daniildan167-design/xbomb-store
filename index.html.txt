        // ------------------------------------------------------
        // 1. НАСТРОЙКИ (ЗАМЕНИТЕ НА СВОИ ДАННЫЕ!)

        const CONFIG = {
            // API-ключ RAWG — 3604decac37e4ec4be13a109597cdbf1

            // Данные Telegram-бота
            BOT_TOKEN: '8824508663:AAHp0cLIbE6KtsIf91hi2r-4Ml9H6Lfk1Co', // от @BotFather
            CHAT_ID: '7892808964', // от @userinfobot
        };

        // ------------------------------------------------------
        // 2. ПЕРЕМЕННЫЕ
        // ------------------------------------------------------
        const gamesGrid = document.getElementById('gamesGrid');
        const searchInput = document.getElementById('searchInput');
        const searchBtn = document.getElementById('searchBtn');
        const modalOverlay = document.getElementById('modalOverlay');
        const modalClose = document.getElementById('modalClose');
        const modalBody = document.getElementById('modalBody');

        let currentGames = []; // Храним все загруженные игры для поиска

        // ------------------------------------------------------
        // 3. ЗАГРУЗКА СПИСКА ИГР С RAWG API
        // ------------------------------------------------------
        async function loadGames(searchQuery = '') {
            try {
                // Платформа Xbox One = 1, Xbox Series X/S = 186
                // Ищем игры с рейтингом > 3.5 для популярных
                let url =
                    `https://api.rawg.io/api/games?key=${CONFIG.RAWG_API_KEY}&platforms=1,186&ordering=-rating&page_size=40`;

                if (searchQuery.trim()) {
                    url =
                        `https://api.rawg.io/api/games?key=${CONFIG.RAWG_API_KEY}&platforms=1,186&search=${encodeURIComponent(searchQuery)}&page_size=40`;
                }

                const response = await fetch(url);
                if (!response.ok) throw new Error('Ошибка загрузки игр');

                const data = await response.json();
                currentGames = data.results || [];

                renderGames(currentGames);

            } catch (error) {
                console.error('Ошибка:', error);
                gamesGrid.innerHTML = `
                        <div style="grid-column:1/-1; text-align:center; padding:60px 20px; color:#888;">
                            <p style="font-size:20px;">⚠️ Не удалось загрузить игры</p>
                            <p style="font-size:14px; margin-top:10px;">Проверьте API-ключ RAWG или интернет-соединение</p>
                            <button onclick="loadGames()" style="margin-top:20px; background:#1ed760; border:none; color:#0a0a0a; padding:12px 30px; border-radius:50px; cursor:pointer; font-weight:700;">
                                Повторить
                            </button>
                        </div>
                    `;
            }
        }

        // ------------------------------------------------------
        // 4. ОТРИСОВКА КАРТОЧЕК
        // ------------------------------------------------------
        function renderGames(games) {
            if (!games || games.length === 0) {
                gamesGrid.innerHTML = `
                        <div style="grid-column:1/-1; text-align:center; padding:60px 20px; color:#888;">
                            <p style="font-size:20px;">🔍 Ничего не найдено</p>
                            <p style="font-size:14px;">Попробуйте изменить запрос</p>
                        </div>
                    `;
                return;
            }

            gamesGrid.innerHTML = games.map(game => {
                const cover = game.background_image || 'https://via.placeholder.com/300x400/1a1a1a/444?text=No+Image';
                const year = game.released ? new Date(game.released).getFullYear() : '—';

                return `
                        <div class="game-card" data-id="${game.id}" onclick="openGameModal(${game.id})">
                            <img src="${cover}" alt="${game.name}" loading="lazy" onerror="this.src='https://via.placeholder.com/300x400/1a1a1a/444?text=Error'">
                            <div class="game-card-info">
                                <h3>${game.name}</h3>
                                <p>${year} ${game.metacritic ? '⭐ ' + game.metacritic : ''}</p>
                            </div>
                        </div>
                    `;
            }).join('');
        }

        // ------------------------------------------------------
        // 5. ОТКРЫТИЕ МОДАЛЬНОГО ОКНА С ДЕТАЛЯМИ
        // ------------------------------------------------------
        async function openGameModal(gameId) {
            try {
                // Показываем загрузку
                modalBody.innerHTML = `
                        <div style="text-align:center; padding:40px 20px; color:#888;">
                            <p>⏳ Загрузка...</p>
                        </div>
                    `;
                modalOverlay.classList.add('active');
                document.body.style.overflow = 'hidden';

                // Запрашиваем детали игры
                const response = await fetch(
                    `https://api.rawg.io/api/games/${gameId}?key=${CONFIG.RAWG_API_KEY}`
                );
                if (!response.ok) throw new Error('Не удалось загрузить детали игры');

                const game = await response.json();

                // Запрашиваем скриншоты
                const screenshotsRes = await fetch(
                    `https://api.rawg.io/api/games/${gameId}/screenshots?key=${CONFIG.RAWG_API_KEY}`
                );
                let screenshots = [];
                if (screenshotsRes.ok) {
                    const screenshotsData = await screenshotsRes.json();
                    screenshots = screenshotsData.results || [];
                }

                // Формируем HTML
                const cover = game.background_image || 'https://via.placeholder.com/600x300/1a1a1a/444?text=No+Image';
                const description = game.description_raw || 'Описание отсутствует.';
                const year = game.released ? new Date(game.released).getFullYear() : '—';
                const genres = game.genres ? game.genres.map(g => g.name).join(', ') : '—';
                const platforms = game.platforms ? game.platforms.map(p => p.platform.name).join(', ') : '—';

                let screenshotsHtml = '';
                if (screenshots.length > 0) {
                    screenshotsHtml = screenshots.map(s =>
                        `<img src="${s.image}" alt="Скриншот" loading="lazy" onerror="this.style.display='none'">`
                    ).join('');
                } else {
                    screenshotsHtml = `<p style="color:#666; font-size:14px;">🖼️ Скриншоты отсутствуют</p>`;
                }

                modalBody.innerHTML = `
                        <img src="${cover}" alt="${game.name}" style="width:100%; max-height:300px; object-fit:cover; border-radius:16px; margin-bottom:10px;">

                        <h2 class="modal-game-title">${game.name}</h2>
                        <div class="modal-game-meta">
                            ${year} • ${genres} • ${platforms}
                            ${game.metacritic ? ' • ⭐ ' + game.metacritic : ''}
                        </div>

                        <div class="modal-game-description">
                            ${description.length > 500 ? description.slice(0, 500) + '...' : description}
                        </div>

                        <h4 style="color:#888; margin:15px 0 8px; font-weight:400;">📸 Скриншоты</h4>
                        <div class="screenshots-container">
                            ${screenshotsHtml}
                        </div>

                        <!-- ФОРМА ПОКУПКИ -->
                        <div class="purchase-form">
                            <h4>🎮 Купить игру</h4>
                            <p style="color:#666; font-size:14px; margin-bottom:15px;">
                                Оставьте свои контакты — мы свяжемся с вами для оформления заказа
                            </p>
                            <form id="purchaseForm" onsubmit="sendOrder(event, ${game.id})">
                                <div class="form-group">
                                    <input type="text" id="userName" placeholder="Ваше имя" required>
                                    <input type="tel" id="userPhone" placeholder="Номер телефона" required>
                                </div>
                                <button type="submit" class="btn-buy" id="buyBtn">📩 Отправить заявку</button>
                                <div class="form-status" id="formStatus"></div>
                            </form>
                        </div>
                    `;

            } catch (error) {
                console.error('Ошибка открытия игры:', error);
                modalBody.innerHTML = `
                        <div style="text-align:center; padding:40px 20px; color:#ff4444;">
                            <p>❌ Ошибка загрузки</p>
                            <p style="font-size:14px; color:#888;">${error.message}</p>
                            <button onclick="modalOverlay.classList.remove('active')" style="margin-top:20px; background:#333; border:none; color:#fff; padding:10px 30px; border-radius:50px; cursor:pointer;">
                                Закрыть
                            </button>
                        </div>
                    `;
            }
        }

        // ------------------------------------------------------
        // 6. ЗАКРЫТИЕ МОДАЛЬНОГО ОКНА
        // ------------------------------------------------------
        function closeModal() {
            modalOverlay.classList.remove('active');
            document.body.style.overflow = '';
        }

        modalClose.addEventListener('click', closeModal);
        modalOverlay.addEventListener('click', (e) => {
            if (e.target === modalOverlay) closeModal();
        });

        // Закрытие по ESC
        document.addEventListener('keydown', (e) => {
            if (e.key === 'Escape') closeModal();
        });

        // ------------------------------------------------------
        // 7. ОТПРАВКА ЗАКАЗА В TELEGRAM
        // ------------------------------------------------------
        async function sendOrder(event, gameId) {
            event.preventDefault();

            const nameInput = document.getElementById('userName');
            const phoneInput = document.getElementById('userPhone');
            const statusDiv = document.getElementById('formStatus');
            const buyBtn = document.getElementById('buyBtn');

            const name = nameInput.value.trim();
            const phone = phoneInput.value.trim();

            if (!name || !phone) {
                statusDiv.className = 'form-status error';
                statusDiv.textContent = '⚠️ Пожалуйста, заполните все поля';
                return;
            }

            // Находим игру по ID
            const game = currentGames.find(g => g.id === gameId);
            const gameName = game ? game.name : 'Игра';

            // Блокируем кнопку
            buyBtn.disabled = true;
            buyBtn.textContent = '⏳ Отправка...';
            statusDiv.className = '';

            try {
                const message = `🆕 НОВЫЙ ЗАКАЗ!\n\n` +
                    `🎮 Игра: ${gameName}\n` +
                    `👤 Имя: ${name}\n` +
                    `📞 Телефон: ${phone}\n` +
                    `🕐 Время: ${new Date().toLocaleString('ru-RU')}`;

                const url = `https://api.telegram.org/bot${CONFIG.BOT_TOKEN}/sendMessage`;

                const response = await fetch(url, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        chat_id: CONFIG.CHAT_ID,
                        text: message,
                        parse_mode: 'HTML',
                    }),
                });

                const result = await response.json();

                if (result.ok) {
                    statusDiv.className = 'form-status success';
                    statusDiv.textContent = '✅ Заявка отправлена! Мы свяжемся с вами в ближайшее время.';
                    nameInput.value = '';
                    phoneInput.value = '';
                } else {
                    throw new Error(result.description || 'Ошибка отправки');
                }

            } catch (error) {
                console.error('Ошибка отправки:', error);
                statusDiv.className = 'form-status error';
                statusDiv.textContent = '❌ Ошибка отправки. Попробуйте ещё раз или свяжитесь с нами напрямую.';
            } finally {
                buyBtn.disabled = false;
                buyBtn.textContent = '📩 Отправить заявку';
            }
        }

        // ------------------------------------------------------
        // 8. ПОИСК
        // ------------------------------------------------------
        function handleSearch() {
            const query = searchInput.value.trim();
            loadGames(query);
        }

        searchBtn.addEventListener('click', handleSearch);
        searchInput.addEventListener('keydown', (e) => {
            if (e.key === 'Enter') handleSearch();
        });

        // ------------------------------------------------------
        // 9. ЗАПУСК
        // ------------------------------------------------------
        // Загружаем игры при открытии страницы
        loadGames();
